pipeline {
agent any
       stages{
        stage('Clone'){
             steps{  
                 git branch: 'raniab', url: 'https://github.com/EmnaFerchichi/DevopsRepos.git'
             }
        }
       
        stage('MVN CLEAN') {
            steps {
                sh 'mvn clean'
                 
            }
        }
           stage("Build") {
                 steps {
sh "mvn compile"
}}

stage("Unit tests") {
steps {
sh "mvn test"
}}
stage('Nexus'){
            steps{
                sh 'mvn deploy -e'
            }
        }
   }
}
